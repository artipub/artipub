DevToPublisherPluginOption ​

ts
interface DevToPublisherPluginOption {
  /**
   * dev.to api key
   */
  api_key: string;
  /**
   * Whether to publish，default: false
   */
  published?: boolean;
  /**
   * Article series
   */
  series?: string;
  /**
   * Article cover map
   */
  main_image?: string;
  /**
   * Article description
   */
  description?: string;
  /**
   * dev.to organization id
   */
  organization_id?: number;
}
